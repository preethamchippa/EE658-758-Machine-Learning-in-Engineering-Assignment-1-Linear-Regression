def fill_missing_values(df, col_name):
    
    if col_name not in df.columns:
        print(f"Error: Column '{col_name}' not found in the DataFrame.")
        return

    if df[col_name].dtype == 'object':  # Categorical column
        mode_value = df[col_name].mode()[0]
        df[col_name].fillna(mode_value, inplace=True)
    else:  # Numeric column
        mean_value = df[col_name].mean()
        df[col_name].fillna(mean_value, inplace=True)

# Fill missing values in the specified column

for col in data.columns:
    fill_missing_values(data, col)

def binary_encoding(df, column_name):
    
    if column_name not in df.columns:
        print(f"Error: Column '{column_name}' not found in the DataFrame.")
        return
    
    unique_values = df[column_name].unique()
    mapping = {value: i for i, value in enumerate(unique_values)}
    df[column_name] = df[column_name].map(mapping)

def one_hot_encoding(df, column_name):
    
    if column_name not in df.columns:
        print(f"Error: Column '{column_name}' not found in the DataFrame.")
        return
    
    # Perform one-hot encoding
    df = pd.get_dummies(df, columns=[column_name], prefix=column_name, drop_first=True)
    
    return df

# Function to train the LinearRegression model with different learning rates and plot the loss curve
def train_and_plot_learning_rate_effect(X_train, y_train, learning_rates):
    plt.figure(figsize=(10, 6))
    for initial_learning_rate in learning_rates:
        # Initialize weights randomly
        alpha = 0.01                  # Initial learning rate
        n_iterations = 10000   # Number of iterations
        decay_factor = 0.1   # Decay factor for learning rate
        decay_interval = 10   # Decay learning rate after every 10 epochs

        w_decay = np.random.randn(D_train+1)  # Weight matrix

        costs_decay = []                          # Loss value for each iteration
        for iteration in range(n_iterations):
            # Calculate gradients
            gradients = 1/N_train * X_bias_train.T.dot(X_bias_train.dot(w_decay) - y_train)

            # Update weights using gradients
            w_decay = w_decay - alpha * gradients

            # Calculate predictions and loss
            predictions = X_bias_train.dot(w_decay)
            costs_decay.append(mean_squared_error(y_train, predictions))

            # Decay learning rate after every decay_interval epochs
            if iteration % decay_interval == 0:
                alpha *= decay_factor

        # Plot the loss curve for the current learning rate
        iterations = np.arange(len(costs_decay))
        plt.plot(iterations, costs_decay, label=f'Learning Rate: {initial_learning_rate}')

    plt.xlabel('Iterations')
    plt.ylabel('Loss')
    plt.title('Effect of Learning Rate on Convergence')
    plt.legend()
    plt.show()

# Define a range of learning rates to test
learning_rates = [0.001, 0.01, 0.1, 0.5, 1.0]

Compare the three solutions in terms of MAE, MSE, and computational efficiency.

Normal Equation:

MSE for test data: 0.7897

Computational Efficiency: Medium

  • The normal equation provides a closed-form solution to linear regression, which can be computationally expensive for large datasets due to matrix inversions. Theoretical Considerations:

  • Scikit-learn: Leverages optimized implementations, handles different loss functions.

  • Gradient Descent: Flexible, requires setting learning rate and number of iterations.

  • Normal Equation: Provides closed-form solution, sensitive to data size and condition number.

Overall:

  • Scikit-learn: Practical choice for good accuracy and efficiency across varied datasets.

  • Gradient Descent: Offers flexibility and potential for optimization, but requires tuning and understanding.

  • Normal Equation: Mainly considered for smaller datasets or theoretical understanding, as its efficiency scales poorly.
